import { z } from 'zod';

export const PostShape = z.object({
  id: z.string().uuid(),
  title: z.string().min(1),
  body: z.string().min(1),
  createdAt: z.date(),
});

export type Post = z.infer<typeof PostShape>;

export function serializePost(post: Post): string {
  return JSON.stringify({ ...post, createdAt: post.createdAt.toISOString() });
}
