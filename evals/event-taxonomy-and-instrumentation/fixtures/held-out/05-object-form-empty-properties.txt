'use server';
import { PostHog } from 'posthog-node';

const posthog = new PostHog(process.env.POSTHOG_KEY!);

export async function recordTutorialSkipped(userId: string) {
  posthog.capture({
    distinctId: userId,
    event: 'tutorial_skipped',
    properties: {},
  });
  await posthog.shutdown();
}
