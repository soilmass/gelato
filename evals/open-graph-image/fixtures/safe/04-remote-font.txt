import { ImageResponse } from 'next/og';

export const size = { width: 1200, height: 630 };
export const contentType = 'image/png';

export default async function OG() {
  const font = await fetch(new URL('./inter.woff2', import.meta.url)).then((r) => r.arrayBuffer());
  return new ImageResponse(
    <div style={{ display: 'flex', fontFamily: 'Inter', fontSize: 96 }}>Gelato</div>,
    { ...size, fonts: [{ name: 'Inter', data: font, weight: 400 }] },
  );
}
