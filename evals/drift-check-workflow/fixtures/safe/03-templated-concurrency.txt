---
name: Drift — Schema.org JSON-LD

on:
  schedule:
    - cron: '15 3 * * 2'
  workflow_dispatch:

concurrency:
  group: ${{ github.workflow }}
  cancel-in-progress: true

jobs:
  drift:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - run: curl -fsSL https://schema.org/version/latest/schemaorg-current-https.jsonld -o /tmp/schema.jsonld
