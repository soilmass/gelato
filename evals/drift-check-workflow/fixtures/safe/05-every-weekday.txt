---
name: Drift — Next.js docs

on:
  schedule:
    - cron: '0 6 * * 1-5'
  workflow_dispatch: {}

concurrency:
  group: drift-nextjs
  cancel-in-progress: true

jobs:
  drift:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - run: curl -fsSL https://nextjs.org/docs -o /tmp/nextjs.html
