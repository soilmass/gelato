import { describe, it, expect } from 'vitest';
import { toSlug } from './slug';

describe('toSlug', () => {
  it('lowercases', () => {
    expect(toSlug('HELLO')).toBe('hello');
  });

  it('hyphenates spaces', () => {
    toSlug('hello world');
  });

  it('trims', () => {
    expect(toSlug('  hello  ')).toBe('hello');
  });
});
