---
class: plaintext-password
reason: Password change endpoint — writes the new password plain via db.update. No hash call.
---
'use server';
import { db, users } from '@/lib/db';
import { eq } from 'drizzle-orm';

export async function changePassword(input: { userId: string; newPassword: string }) {
  await db.update(users)
    .set({ password: input.newPassword })
    .where(eq(users.id, input.userId));
  return { ok: true };
}
