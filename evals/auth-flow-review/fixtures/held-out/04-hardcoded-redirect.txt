import { redirect } from 'next/navigation';

export default async function PostLogin({
  searchParams,
}: {
  searchParams: { toast?: string };
}) {
  const toast = searchParams.toast ?? '';
  if (toast === 'welcome') {
    console.log('welcome-toast shown');
  }
  redirect('/dashboard');
}
