'use server';
import { db, users } from '@/lib/db';

export async function saveSeededUser(input: { email: string; passwordHash: string }) {
  await db.insert(users).values({
    email: input.email,
    passwordHash: input.passwordHash,
  });
  return { ok: true };
}
