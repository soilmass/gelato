---
expected: safe
reason: Cookie set is for UI preferences (theme), not session/auth. Classifier only enforces flags on auth-named cookies.
---
'use server';
import { cookies } from 'next/headers';

export async function setTheme(theme: 'light' | 'dark') {
  cookies().set('theme', theme, {
    path: '/',
    maxAge: 60 * 60 * 24 * 365,
  });
  return { ok: true };
}
