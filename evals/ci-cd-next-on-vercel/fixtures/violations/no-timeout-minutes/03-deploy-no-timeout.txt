---
class: no-timeout-minutes
reason: Production deploy job with no timeout. A stuck deploy runs for 6 hours before GitHub reaps it.
---
name: Deploy (production)
on:
  push:
    branches: [main]

concurrency:
  group: deploy-prod
  cancel-in-progress: false

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: oven-sh/setup-bun@v1
      - run: bun install --frozen-lockfile
      - name: Deploy
        env:
          VERCEL_TOKEN: ${{ secrets.VERCEL_TOKEN }}
        run: bunx vercel deploy --prod --token="$VERCEL_TOKEN"
