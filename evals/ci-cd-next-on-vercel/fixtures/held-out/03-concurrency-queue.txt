---
class: safe
reason: "Production workflow uses cancel-in-progress: false to queue deploys — legitimate, not a violation."
---
name: Deploy (production)
on:
  push:
    branches: [main]

concurrency:
  group: prod
  cancel-in-progress: false

jobs:
  deploy:
    runs-on: ubuntu-latest
    timeout-minutes: 15
    steps:
      - uses: actions/checkout@v4
      - uses: oven-sh/setup-bun@v1
      - run: bun install --frozen-lockfile
      - name: Deploy
        env:
          VERCEL_TOKEN: ${{ secrets.VERCEL_TOKEN }}
        run: bunx vercel deploy --prod --token="$VERCEL_TOKEN"
