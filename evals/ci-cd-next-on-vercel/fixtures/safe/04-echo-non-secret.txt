---
expected: safe
reason: Echo of non-secret values (workflow name, commit SHA) is fine. No secret referenced in echo.
---
name: CI
on: pull_request

concurrency:
  group: ci-${{ github.ref }}
  cancel-in-progress: true

jobs:
  lint:
    runs-on: ubuntu-latest
    timeout-minutes: 5
    steps:
      - uses: actions/checkout@v4
      - run: echo "Running CI on commit ${{ github.sha }}"
      - run: echo "Branch is ${{ github.ref }}"
      - uses: oven-sh/setup-bun@v1
      - run: bun install --frozen-lockfile
      - run: bun run lint
