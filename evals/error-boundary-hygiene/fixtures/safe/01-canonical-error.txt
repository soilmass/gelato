'use client';
export default function Error({
  error,
  reset,
}: {
  error: Error & { digest?: string };
  reset: () => void;
}) {
  return (
    <div>
      <h2>Something went wrong in the blog</h2>
      {error.digest ? <p>Reference: {error.digest}</p> : null}
      <button onClick={() => reset()}>Try again</button>
    </div>
  );
}
