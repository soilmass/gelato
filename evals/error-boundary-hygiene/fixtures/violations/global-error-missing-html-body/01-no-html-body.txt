'use client';
export default function GlobalError({
  error,
  reset,
}: {
  error: Error;
  reset: () => void;
}) {
  return (
    <div>
      <h2>App crashed</h2>
      <button onClick={() => reset()}>Reload</button>
    </div>
  );
}
