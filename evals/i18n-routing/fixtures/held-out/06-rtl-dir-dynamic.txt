const RTL = new Set(['ar', 'he', 'fa', 'ur']);

export default function Layout({
  children,
  params,
}: {
  children: React.ReactNode;
  params: { locale: string };
}) {
  return (
    <html lang={params.locale} dir={RTL.has(params.locale) ? 'rtl' : 'ltr'}>
      <body>{children}</body>
    </html>
  );
}
