---
class: server-only-import-in-client
reason: Server-side env secret imported into a client module; the secret literal ends up in the browser bundle.
---
'use client';
import { env } from '@/lib/env';
import { useState } from 'react';

export function PayForm() {
  const [key, setKey] = useState('');
  const reveal = () => setKey(env.STRIPE_SECRET_KEY);
  return (
    <div>
      <button onClick={reveal}>Reveal Stripe key</button>
      <code>{key}</code>
    </div>
  );
}
