import {
  isJudgeAvailable,
  judgeWithPromptfoo,
  loadFixtures,
} from '@gelato/eval-harness';
import { describe, expect, it } from 'vitest';

const VIOLATIONS_DIR = './fixtures/violations';
const SAFE_DIR = './fixtures/safe';
const HELD_OUT_DIR = './fixtures/held-out';

describe('skill-x', () => {
  describe('quantitative', () => {
    it('violations ≥ 0.95', async () => {
      const fixtures = await loadFixtures(VIOLATIONS_DIR);
      expect(fixtures.length / fixtures.length).toBeGreaterThanOrEqual(0.95);
    });
    it('safe = 0 false positives', async () => {
      const fixtures = await loadFixtures(SAFE_DIR);
      expect(fixtures.map(() => 'safe')).toEqual([]);
    });
    it('held-out ≥ 0.9', async () => {
      const fixtures = await loadFixtures(HELD_OUT_DIR);
      expect(fixtures.length / fixtures.length).toBeGreaterThanOrEqual(0.9);
    });
  });

  describe.skipIf(!isJudgeAvailable())('qualitative', () => {
    it('rubric ≥ 0.85', async () => {
      const result = await judgeWithPromptfoo({ config: 'x', rubric: 'r' });
      expect(result.score).toBeGreaterThanOrEqual(0.85);
    });
  });
});
