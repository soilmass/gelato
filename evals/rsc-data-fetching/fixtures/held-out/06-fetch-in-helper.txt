async function loadUsers() {
  return fetch('https://api.example.com/users', {
    next: { revalidate: 120, tags: ['users'] },
  }).then((r) => r.json());
}

export default async function Page() {
  const users = await loadUsers();
  return <UserList users={users} />;
}
