---
class: sensitive-data-logged
reason: Logs apiKey and secret — either one is a breach, together they are catastrophic.
---
import { logger } from '@/lib/logger';

export async function POST(req: Request) {
  const { apiKey, secret } = await req.json();
  logger.warn({ apiKey, secret, route: '/webhook/rotate' }, 'key rotation request');
  return Response.json({ ok: true });
}
