'use client';
import { useState } from 'react';
import { useActionState } from 'react';
import { savePreferences } from './actions';

export function PreferencesForm() {
  const [theme, setTheme] = useState('light');
  const [state, formAction, pending] = useActionState(savePreferences, { ok: false, errors: {} });

  return (
    <form action={formAction}>
      <input name="theme" type="radio" value="light" checked={theme === 'light'} onChange={() => setTheme('light')} />
      <input name="theme" type="radio" value="dark" checked={theme === 'dark'} onChange={() => setTheme('dark')} />
      {state.errors?.theme?.length ? <p role="alert">{state.errors.theme.join(', ')}</p> : null}
      <button type="submit" disabled={pending}>Save</button>
    </form>
  );
}
