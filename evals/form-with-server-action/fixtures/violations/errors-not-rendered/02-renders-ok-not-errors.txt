---
class: errors-not-rendered
reason: JSX renders a success message but no error display. Validation failures disappear silently.
---
'use client';
import { useActionState } from 'react';
import { contact } from './actions';

export function ContactForm() {
  const [state, formAction, pending] = useActionState(contact, {
    ok: false,
    errors: { email: [], body: [] },
  });

  return (
    <>
      {state.ok ? <p className="text-green-600">Thanks — we'll be in touch.</p> : null}
      <form action={formAction}>
        <input name="email" type="email" defaultValue="" required />
        <textarea name="body" defaultValue="" required />
        <button type="submit" disabled={pending}>Send</button>
      </form>
    </>
  );
}
