---
class: action-not-bound-to-form
reason: File imports a Server Action but calls it from onSubmit with a manual FormData build — no action={} attribute, binding bypassed.
---
'use client';
import { contact } from './actions';

export function ContactForm() {
  return (
    <form
      onSubmit={async (e) => {
        e.preventDefault();
        const formData = new FormData(e.currentTarget);
        await contact(formData);
      }}
    >
      <input name="email" type="email" defaultValue="" required />
      <textarea name="body" defaultValue="" required />
      <button type="submit">Send</button>
    </form>
  );
}
