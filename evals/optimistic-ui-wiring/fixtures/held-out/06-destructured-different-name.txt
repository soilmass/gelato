'use client';
import { useOptimistic } from 'react';

export function List({ items }: { items: string[] }) {
  const [optimistic, enqueue] = useOptimistic(
    items,
    (state, v: string) => [...state, v],
  );
  return <button onClick={() => enqueue('new')}>{optimistic.length}</button>;
}
