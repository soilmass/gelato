import * as Sentry from '@sentry/nextjs';
import { logger } from '@/lib/logger';

export async function fetchRemote<T>(url: string): Promise<T | null> {
  try {
    const res = await fetch(url);
    return (await res.json()) as T;
  } catch (err) {
    logger.warn({ err, url }, 'fetchRemote failed');
    return null;
  }
}

void Sentry;
